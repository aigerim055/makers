trl + alt + t - открыть терминал

ls - просмотр содержимого текущей директории
ls путь_до_директории - просмотр содержимого указанной директории
ls -a - просмотр скрытых файлов директории

mkdir название_директории - создание директории
cd путь_до_директории - смена директории

"." - ссылка на текущую директорию
".." - ссылка на директорию выше/родительскую
"~" - ссылка на домашнюю директорию

pwd - вывод абсолютного пути в консоль

абсолютный путь - путь, начинающийся с домашней директории (home/user/some_dir)
относительный путь - путь, относительно других директорий (./some_dir)

tree

rmdir путь_до_директории - удаление пустой директории
rm -rf имя_директории - принудительное удаление директории с содержимым

touch - создание файлов в текущей директории
rm имя_файла - удаление файлов
rm -rf имя_файла - принудительное удаление файла

cat имя_файла - вывод содержимого файла в терминал
cat имя_файла >> имя_другого_файла - дозапись содержмиого одного файла в другой

echo "какой-то текст" - вывод указанного текста в терминал
echo "какой-то текст" >> название_файла - дозапись указанного текста в файл

mv название_файла новое_название_файла - переименование
mv название_файла путь_до_директории - перенос файла в указанную директорию


nano путь_до_файла - открытие файла с помощью текстового редактора nano. Если указанного файла не существует, то создает его


первая_команда && последующая_команда - && - объединение команд
